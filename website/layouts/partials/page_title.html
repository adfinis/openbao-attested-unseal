{{- $title := .Params.title -}} {{- if not $title -}} {{- $matches := findRE `(?m)^#\s+(.+)$` .RawContent 1 -}} {{- if gt (len $matches) 0 -}} {{- $title = index $matches 0 | replaceRE `(?m)^#\s+` "" | markdownify | plainify | strings.TrimSpace -}} {{- else if .LinkTitle -}} {{- $title = .LinkTitle -}} {{- else if .File -}} {{- $title = humanize .File.TranslationBaseName -}} {{- else -}} {{- $title = .Title -}} {{- end -}} {{- end -}} {{- return $title -}}
